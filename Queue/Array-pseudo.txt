Class QueueArray:
    max_size = 20
    front = 0
    rear = -1
    data[20]  # array to store queue elements

# Check if Queue is empty
Function isEmpty():
    if front > rear:
        return True
    else:
        return False

# Check if Queue is full
Function isFull():
    if rear == max_size - 1:
        return True
    else:
        return False

# Enqueue operation
Function enqueue(item):
    if isFull():
        print("Queue Overflow - Cannot enqueue", item)
    else:
        rear = rear + 1
        data[rear] = item
        print("Enqueued:", item)

# Dequeue operation
Function dequeue():
    if isEmpty():
        print("Queue Underflow - Queue is empty")
        return None
    else:
        item = data[front]
        print("Dequeued:", item)
        front = front + 1
        return item

# Display Queue elements
Function display():
    if isEmpty():
        print("Queue is empty")
    else:
        for i from front to rear:
            print(data[i])
