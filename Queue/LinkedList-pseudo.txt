Class Node
    data
    link


Class QueueLinkedList
    front = NULL
    rear = NULL


=========================
Function: isEmpty()
Precondition: Queue is initialized
Postcondition: Returns True if queue is empty, else False
Time Complexity: O(1)
Edge Case: Returns True if both front and rear are NULL
=========================
Function isEmpty():
    if front == NULL:
        return True
    else:
        return False


=========================
Function: create_node(data)
Precondition: Data is valid
Postcondition: Returns a new node containing the given data
Time Complexity: O(1)
Edge Case: None (assumes valid data)
=========================
Function create_node(data):
    node = new Node
    node.data = data
    node.link = NULL
    return node


=========================
Function: enqueue(item)
Precondition: Queue is initialized
Postcondition: Inserts a new node at the rear of the queue
Time Complexity: O(1)
Edge Cases: 
- If queue is empty, front and rear both point to the new node
=========================
Function enqueue(item):
    temp = create_node(item)
    if rear == NULL:  # Queue is empty
        front = rear = temp
    else:
        rear.link = temp
        rear = temp
    print("Enqueued:", item)


=========================
Function: dequeue()
Precondition: Queue is initialized
Postcondition: Removes and returns the element at the front of the queue
Time Complexity: O(1)
Edge Cases:
- Underflow if queue is empty
- If dequeued element was the last node, update both front and rear to NULL
=========================
Function dequeue():
    if isEmpty():
        print("Queue Underflow - Queue is empty")
        return None
    else:
        item = front.data
        front = front.link
        if front == NULL:  # Queue became empty
            rear = NULL
        print("Dequeued:", item)
        return item


=========================
Function: display()
Precondition: Queue is initialized
Postcondition: Prints all elements from front to rear
Time Complexity: O(n) where n = number of elements in queue
Edge Case: If queue is empty, prints "Queue is empty"
=========================
Function display():
    if isEmpty():
        print("Queue is empty")
    else:
        temp = front
        while temp != NULL:
            print(temp.data)
            temp = temp.link
